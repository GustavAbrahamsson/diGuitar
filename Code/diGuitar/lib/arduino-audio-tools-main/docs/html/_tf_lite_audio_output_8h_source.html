103 // Partial implementation of std::dequeue, just providing the functionality
104 // that's needed to keep a record of previous neural network results over a
105 // short time period, so they can be averaged together to produce a more
106 // accurate overall prediction. This doesn't use any dynamic memory allocation
107 // so it's a better fit for microcontroller applications, but this does mean
108 // there are hard limits on the number of results it can store.
109 template <int N>
110 class TfLiteResultsQueue {
111  public:
112  TfLiteResultsQueue() : front_index_(0), size_(0) {}
113 
114  // Data structure that holds an inference result, and the time when it
115  // was recorded.
116  struct Result {
117  Result() : time_(0), scores() {}
118  Result(int32_t time, int8_t* input_scores) : time_(time) {
119  for (int i = 0; i < N; ++i) {
120  scores[i] = input_scores[i];
121  }
122  }
123  int32_t time_;
124  int8_t scores[N];
125  };
126 
127  int size() { return size_; }
128  bool empty() { return size_ == 0; }
129  Result& front() { return results_[front_index_]; }
130  Result& back() {
131  int back_index = front_index_ + (size_ - 1);
132  if (back_index >= kMaxResults) {
133  back_index -= kMaxResults;
134  }
135  return results_[back_index];
136  }
137 
138  void push_back(const Result& entry) {
139  if (size() >= kMaxResults) {
140  LOGE("Couldn't push_back latest result, too many already!");
141  return;
142  }
143  size_ += 1;
144  back() = entry;
145  }
146 
147  Result pop_front() {
148  if (size() <= 0) {
149  LOGE("Couldn't pop_front result, none present!");
150  return Result();
151  }
152  Result result = front();
153  front_index_ += 1;
154  if (front_index_ >= kMaxResults) {
155  front_index_ = 0;
156  }
157  size_ -= 1;
158  return result;
159  }
160 
161  // Most of the functions are duplicates of dequeue containers, but this
162  // is a helper that makes it easy to iterate through the contents of the
163  // queue.
164  Result& from_front(int offset) {
165  if ((offset < 0) || (offset >= size_)) {
166  LOGE("Attempt to read beyond the end of the queue!");
167  offset = size_ - 1;
168  }
169  int index = front_index_ + offset;
170  if (index >= kMaxResults) {
171  index -= kMaxResults;
172  }
173  return results_[index];
174  }
175 
176  private:
177  static constexpr int kMaxResults = 50;
178  Result results_[kMaxResults];
179 
180  int front_index_;
181  int size_;
182 };

216 class TfLiteRecognizeCommands : public TfLiteAbstractRecognizeCommands<N> {
217  public:
218  // labels should be a list of the strings associated with each one-hot score.
219  // The window duration controls the smoothing. Longer durations will give a
220  // higher confidence that the results are correct, but may miss some commands.
221  // The detection threshold has a similar effect, with high values increasing
222  // the precision at the cost of recall. The minimum count controls how many
223  // results need to be in the averaging window before it's seen as a reliable
224  // average. This prevents erroneous results when the averaging window is
225  // initially being populated for example. The suppression argument disables
226  // further recognitions for a set time after one has been triggered, which can
227  // help reduce spurious recognitions.
228 
229  explicit TfLiteRecognizeCommands() {
230  previous_top_label_ = "silence";
231  previous_top_label_time_ = std::numeric_limits<int32_t>::min();
232  kCategoryCount = N;
233  }
234 
236  bool begin(TfLiteConfig cfg) override {
237  average_window_duration_ms_ = cfg.average_window_duration_ms;
238  detection_threshold_ = cfg.detection_threshold;
239  suppression_ms_ = cfg.suppression_ms;
240  minimum_count_ = cfg.minimum_count;
241  kCategoryLabels = cfg.labels;
242  if (cfg.labels==0){
243  LOGW("config.labels not defined");
244  return false;
245  }
246  return true;
247  }

249  // Call this with the results of running a model on sample data.
250  virtual TfLiteStatus ProcessLatestResults(const TfLiteTensor* latest_results,
251  const int32_t current_time_ms,
252  const char** found_command, uint8_t* score,
253  bool* is_new_command) override {
254  LOGD(LOG_METHOD);
255  if ((latest_results->dims->size != 2) ||
256  (latest_results->dims->data[0] != 1) ||
257  (latest_results->dims->data[1] != kCategoryCount)) {
258  LOGE(
259  "The results for recognition should contain %d "
260  "elements, but there are "
261  "%d in an %d-dimensional shape",
262  kCategoryCount, (int)latest_results->dims->data[1],
263  (int)latest_results->dims->size);
264  return kTfLiteError;
265  }
266 
267  if (latest_results->type != kTfLiteInt8) {
268  LOGE("The results for recognition should be int8 elements, but are %d",
269  (int)latest_results->type);
270  return kTfLiteError;
271  }
272 
273  if ((!previous_results_.empty()) &&
274  (current_time_ms < previous_results_.front().time_)) {
275  LOGE("Results must be in increasing time order: timestamp %d < %d",
276  (int)current_time_ms, (int)previous_results_.front().time_);
277  return kTfLiteError;
278  }
279 
280  // Add the latest results to the head of the queue.
281  previous_results_.push_back({current_time_ms, latest_results->data.int8});
282 
283  // Prune any earlier results that are too old for the averaging window.
284  const int64_t time_limit = current_time_ms - average_window_duration_ms_;
285  while ((!previous_results_.empty()) &&
286  previous_results_.front().time_ < time_limit) {
287  previous_results_.pop_front();
288  }
289 
290  // If there are too few results, assume the result will be unreliable
291  // and bail.
292  const int64_t how_many_results = previous_results_.size();
293  const int64_t earliest_time = previous_results_.front().time_;
294  const int64_t samples_duration = current_time_ms - earliest_time;
295  if ((how_many_results < minimum_count_) ||
296  (samples_duration < (average_window_duration_ms_ / 4))) {
297  *found_command = previous_top_label_;
298  *score = 0;
299  *is_new_command = false;
300  return kTfLiteOk;
301  }
302 
303  // Calculate the average score across all the results in the window.
304  int32_t average_scores[kCategoryCount];
305  for (int offset = 0; offset < previous_results_.size(); ++offset) {
306  auto previous_result = previous_results_.from_front(offset);
307  const int8_t* scores = previous_result.scores;
308  for (int i = 0; i < kCategoryCount; ++i) {
309  if (offset == 0) {
310  average_scores[i] = scores[i] + 128;
311  } else {
312  average_scores[i] += scores[i] + 128;
313  }
314  }
315  }
316  for (int i = 0; i < kCategoryCount; ++i) {
317  average_scores[i] /= how_many_results;
318  }
319 
320  // Find the current highest scoring category.
321  int current_top_index = 0;
322  int32_t current_top_score = 0;
323  for (int i = 0; i < kCategoryCount; ++i) {
324  if (average_scores[i] > current_top_score) {
325  current_top_score = average_scores[i];
326  current_top_index = i;
327  }
328  }
329  const char* current_top_label = kCategoryLabels[current_top_index];
330 
331  // If we've recently had another label trigger, assume one that occurs
332  // too soon afterwards is a bad result.
333  int64_t time_since_last_top;
334  if ((previous_top_label_ == kCategoryLabels[0]) ||
335  (previous_top_label_time_ == std::numeric_limits<int32_t>::min())) {
336  time_since_last_top = std::numeric_limits<int32_t>::max();
337  } else {
338  time_since_last_top = current_time_ms - previous_top_label_time_;
339  }
340  if ((current_top_score > detection_threshold_) &&
341  ((current_top_label != previous_top_label_) ||
342  (time_since_last_top > suppression_ms_))) {
343  previous_top_label_ = current_top_label;
344  previous_top_label_time_ = current_time_ms;
345  *is_new_command = true;
346  } else {
347  *is_new_command = false;
348  }
349  *found_command = current_top_label;
350  *score = current_top_score;
351 
352  return kTfLiteOk;
353  }
354 
355  protected:
356  // Configuration
357  int32_t average_window_duration_ms_;
358  uint8_t detection_threshold_;
359  int32_t suppression_ms_;
360  int32_t minimum_count_;
361  int kCategoryCount;
362  const char** kCategoryLabels = nullptr;
363 
364  // Working variables
365  TfLiteResultsQueue<N> previous_results_;
366  const char* previous_top_label_;
367  int32_t previous_top_label_time_;
368 };

526  virtual TfLiteStatus generateMicroFeatures(const int16_t* input, int input_size,
527  int output_size, int8_t* output,
528  size_t* num_samples_read) {
529  LOGD(LOG_METHOD);
530  const int16_t* frontend_input;
531  if (g_is_first_time) {
532  frontend_input = input;
533  g_is_first_time = false;
534  } else {
535  frontend_input = input;
536  }
537 
538  // Apply FFT
539  FrontendOutput frontend_output = FrontendProcessSamples(
540  &g_micro_features_state, frontend_input, input_size, num_samples_read);
541 
542  for (size_t i = 0; i < frontend_output.size; ++i) {
543  // These scaling values are derived from those used in input_data.py in
544  // the training pipeline. The feature pipeline outputs 16-bit signed
545  // integers in roughly a 0 to 670 range. In training, these are then
546  // arbitrarily divided by 25.6 to get float values in the rough range of
547  // 0.0 to 26.0. This scaling is performed for historical reasons, to match
548  // up with the output of other feature generators. The process is then
549  // further complicated when we quantize the model. This means we have to
550  // scale the 0.0 to 26.0 real values to the -128 to 127 signed integer
551  // numbers. All this means that to get matching values from our integer
552  // feature output into the tensor input, we have to perform: input =
553  // (((feature / 25.6) / 26.0) * 256) - 128 To simplify this and perform it
554  // in 32-bit integer math, we rearrange to: input = (feature * 256) /
555  // (25.6 * 26.0) - 128
556  constexpr int32_t value_scale = 256;
557  constexpr int32_t value_div =
558  static_cast<int32_t>((25.6f * 26.0f) + 0.5f);
559  int32_t value =
560  ((frontend_output.values[i] * value_scale) + (value_div / 2)) /
561  value_div;
562  value -= 128;
563  if (value < -128) {
564  value = -128;
565  }
566  if (value > 127) {
567  value = 127;
568  }
569  output[i] = value;
570  }
571 
572  return kTfLiteOk;
573  }

780  size_t processAudio(const uint8_t* audio, size_t bytes) {
781  LOGD("process: %u", (unsigned)bytes);
782  // Update the spectrogram
783  total_slice_count += feature_provider->write(audio, bytes);
784 
785  // run network model only if we have the necessary slices
786  if (total_slice_count < cfg.kSlicesToProcess) {
787  return bytes;
788  }
789 
790  LOGI("->slices: %d", total_slice_count);
791  // Copy feature buffer to input tensor
792  // for (int i = 0; i < cfg.featureElementCount(); i++) {
793  // model_input_buffer[i] = feature_buffer[i];
794  // }
795  memcpy(model_input_buffer, feature_buffer, cfg.featureElementCount());
796

798  TfLiteStatus invoke_status = interpreter->Invoke();
799  if (invoke_status != kTfLiteOk) {
800  LOGE("Invoke failed");
801  return 0;
802  }
803 
804  // Obtain a pointer to the output tensor
805  TfLiteTensor* output = interpreter->output(0);
806 
807  // determine time
808  current_time += cfg.kFeatureSliceStrideMs * total_slice_count;
809  // Determine whether a command was recognized
810  const char* found_command = nullptr;
811  uint8_t score = 0;
812  bool is_new_command = false;
813 
814  TfLiteStatus process_status = recognizer->ProcessLatestResults(
815  output, current_time, &found_command, &score, &is_new_command);
816  if (process_status != kTfLiteOk) {
817  LOGE("TfLiteRecognizeCommands::ProcessLatestResults() failed");
818  return 0;
819  }
820  // Do something based on the recognized command. The default
821  // implementation just prints to the error console, but you should replace
822  // this with your own function for a real application.
823  respondToCommand(found_command, score, is_new_command);
824 
825  // reset total_slice_count
826  total_slice_count = 0;
827 
828  // all bytes processed successfully
829  return bytes;
830  }
