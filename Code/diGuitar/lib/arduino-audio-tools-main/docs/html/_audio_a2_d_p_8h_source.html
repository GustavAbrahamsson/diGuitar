313  // callback used by A2DP to provide the a2dp_source sound data
314  static int32_t a2dp_stream_source_sound_data(Frame* data, int32_t len) {
315  if (a2dp_buffer==nullptr) return 0;
316  int32_t result_len = 0;
317 
318  bool isAvailable = false;
319  bool isLocked = false;
320  if (is_a2dp_active) {
321  if (A2DPStream_self->lockSemaphore(true)) {
322  isAvailable = a2dp_buffer->available()>0;
323  LOGD("buffer: %d, free %d",a2dp_buffer->available(), a2dp_buffer->availableForWrite() )
324  isLocked = true;
325  } else {
326  LOGE("lock failed");
327  }
328  }
329 
330  // at first call we start with some empty data
331  if (isAvailable){
332  // the data in the file must be in int16 with 2 channels
333  size_t result_len_bytes = a2dp_buffer->readArray((uint8_t*)data, len*sizeof(Frame));
334  A2DPStream_self->lockSemaphore(false);
335 
336  // result is in number of frames
337  result_len = result_len_bytes / sizeof(Frame);
338  } else {
339  if (isLocked) A2DPStream_self->lockSemaphore(false);
340 
341  // prevent underflow on first call
342  switch (A2DPStream_self->config.noData) {
343  case A2DPSilence:
344  memset(data, 0, len*sizeof(Frame));
345  break;
346  case A2DPWhoosh:
347  for (int j=0;j<len;j++){
348  data[j].channel2 = data[j].channel1 = (rand() % 50) - 25;
349  }
350  break;
351  }

357 
358  }
