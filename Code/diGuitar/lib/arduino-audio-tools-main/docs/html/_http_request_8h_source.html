176  is_ready = false;

180  }

184  if (!is_connected){

186  return -1;
187  }
188  } else {
189  LOGI("process is already connected");
190  }
191 
192 #ifdef ESP32
193  LOGI("Free heap: %u", ESP.getFreeHeap());
194 #endif
195 
196  host_name = url.host();
197  request_header.setValues(action, url.path());
198  if (len>0 && data!=nullptr){
199  len = strlen(data);
200  request_header.put(CONTENT_LENGTH, len);
201  }
202  request_header.put(HOST_C, host_name);
203  request_header.put(CONNECTION, connection);
204  request_header.put(USER_AGENT, agent);
205  request_header.put(ACCEPT_ENCODING, accept_encoding);
206  request_header.put(ACCEPT, accept);
207  request_header.put(CONTENT_TYPE, mime);
208 
209  request_header.write(*client_ptr);
210 
211  if (len>0){
212  LOGI("Writing data: %d bytes", len);
213  client_ptr->write((const uint8_t*)data,len);
214  LOGD("%s",data);
215  }
216 
217  LOGI("Request written ... waiting for reply")
218  client_ptr->flush();
219  reply_header.read(*client_ptr);

221  // if we use chunked tranfer we need to read the first chunked length

224  };
