128  static void task(void * pvParameters){

130  static uint8_t buffer[512];

132  while(true){
133  size_t available_to_write = self->buffers.availableForWrite();
134  if (*(self->p_stream) && available_to_write>0){
135  size_t to_read = min(available_to_write, (size_t) 512);
136  size_t avail_read = self->p_stream->readBytes((uint8_t*)buffer, to_read);
137  xSemaphoreTake(self->mutex, portMAX_DELAY);
138  size_t written = self->buffers.writeArray(buffer, avail_read);
139  xSemaphoreGive(self->mutex);
