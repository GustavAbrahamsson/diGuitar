66  audio_hal_iface_samples_t toSampleRate() {
67  LOGD(LOG_METHOD);
68  const static int ia[] = {8000, 11025, 16000, 22050,
69  24000, 32000, 44100, 48000};
70  const static audio_hal_iface_samples_t oa[] = {
71  AUDIO_HAL_08K_SAMPLES, AUDIO_HAL_11K_SAMPLES, AUDIO_HAL_16K_SAMPLES,
72  AUDIO_HAL_22K_SAMPLES, AUDIO_HAL_24K_SAMPLES, AUDIO_HAL_32K_SAMPLES,
73  AUDIO_HAL_44K_SAMPLES, AUDIO_HAL_48K_SAMPLES};
74  int diff = 99999;
75  int result = 0;
76  for (int j = 0; j < 8; j++) {
77  if (ia[j] == sample_rate) {
78  LOGD("-> %d",ia[j])
79  return oa[j];
80  } else {
81  int new_diff = abs(oa[j] - sample_rate);
82  if (new_diff < diff) {
83  result = j;
84  diff = new_diff;
85  }
86  }
87  }
88  LOGE("Sample Rate not supported: %d - using %d", sample_rate, ia[result]);
89  return oa[result];
90  }

222  virtual size_t readBytes(uint8_t *data, size_t length) override {
223  if (cfg.channels == 2) {
224  return kit.read(data, length);
225  } else if (cfg.channels==1) {
226  // convert 2 channels of int16_t to 1
227  int16_t temp[length];
228  int len_res = kit.read(temp, length*2);
229  int res_count = len_res / 2;
230  int16_t *out = (int16_t*) data;
231  for (int j=0; j<res_count; j+=2){
232  int32_t total = temp[j];
233  total+=temp[j+1];
234  total = total/2;
235  *out = total;
236  out++;
237  }
238  return res_count;
239  }
240  LOGE("Unsuported number of channels : %d", cfg.channels);
241  return 0;
