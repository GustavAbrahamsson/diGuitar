184  // find closest frequency for requested sample_rate
185  float freq_requested = cfg.sample_rate * cfg.bits_per_sample ;
186  float selected_freq = 0;
187  for (auto freq : freq_table) {
188  for (auto div : ratio_table) {
189  float freq_value = freq.freq * 1000000 / div.ratio;
190  if (abs(freq_value-freq_requested) < abs(selected_freq-freq_requested)){
191  // MCKFREQ
192  NRF_I2S->CONFIG.MCKFREQ = freq.id << I2S_CONFIG_MCKFREQ_MCKFREQ_Pos;
193  // Ratio
194  NRF_I2S->CONFIG.RATIO = div.id << I2S_CONFIG_RATIO_RATIO_Pos;
195  selected_freq = freq_value;
196  }
197  }
198  LOGI("frequency requested %f vs %f", freq_requested, selected_freq);
199  }
