518  bool setVolume(int mode, int volume, int dot) {
519  LOGI("setVolume(%d,%d,%d)", mode, volume, dot);
520  int res = 0;
521  if (volume < -96 || volume > 0) {
522  LOGW("Warning: volume < -96! or > 0!\n");
523  if (volume < -96)
524  volume = -96;
525  else
526  volume = 0;
527  }
528  dot = (dot >= 5 ? 1 : 0);
529  volume = (-volume << 1) + dot;
530  if (mode == ES_MODULE_ADC || mode == ES_MODULE_ADC_DAC) {
531  res |= i2c_write_reg(ES8388_ADCCONTROL8, volume);
532  // ADC Right Volume=0db
533  res |= i2c_write_reg(ES8388_ADCCONTROL9, volume);
534  }
535  if (mode == ES_MODULE_DAC || mode == ES_MODULE_ADC_DAC) {
536  res |= i2c_write_reg(ES8388_DACCONTROL5, volume);
537  res |= i2c_write_reg(ES8388_DACCONTROL4, volume);
538  }
539  return res == 0;
540  }

880  bool initES8388(bool isMaster, audio_hal_dac_output_t dac,
881  audio_hal_adc_input_t adc) {
882  LOGD(LOG_METHOD);
883  int res = 0;
884 
885  res = i2c_init(); // ESP32 in master mode
886  // 0x04 mute/0x00 unmute&ramp;DAC unmute and disabled digital volume control
887  // soft ramp
888  res |= i2c_write_reg(ES8388_DACCONTROL3, 0x04);
889  // Chip Control and Power Management
890  res |= i2c_write_reg(ES8388_CONTROL2, 0x50);
891  // normal all and power up all
892  res |= i2c_write_reg(ES8388_CHIPPOWER, 0x00);
893 
894  // Disable the internal DLL to improve 8K sample rate
895  res |= i2c_write_reg(0x35, 0xA0);
896  res |= i2c_write_reg(0x37, 0xD0);
897  res |= i2c_write_reg(0x39, 0xD0);
898 
899  // CODEC IN I2S SLAVE MODE
900  res |= i2c_write_reg(ES8388_MASTERMODE,
901  isMaster ? ES_MODE_MASTER : ES_MODE_SLAVE);
902 
903  // disable DAC and disable Lout/Rout/1/2
904  res |= i2c_write_reg(ES8388_DACPOWER, 0xC0);
905  // Enfr=0,Play&Record Mode,(0x17-both of mic&paly)
906  res |= i2c_write_reg(ES8388_CONTROL2, 0);
907  // LPVrefBuf=0,Pdn_ana=0
908  res |= i2c_write_reg(ES8388_CONTROL1, 0x12);
909 
910  // DAC
911  // 1a 0x18:16bit iis , 0x00:24
912  res |= i2c_write_reg(ES8388_DACCONTROL1, 0x18);
913  // DACFsMode,SINGLE SPEED; DACFsRatio,256
914  res |= i2c_write_reg(ES8388_DACCONTROL2, 0x02);
915  // 0x00 audio on LIN1&RIN1,
916  res |= i2c_write_reg(ES8388_DACCONTROL16, 0x00);
917  // only left DAC to left mixer enable 0db
918  res |= i2c_write_reg(ES8388_DACCONTROL17, 0x90);
919  // only right DAC to right mixer enable 0db
920  res |= i2c_write_reg(ES8388_DACCONTROL20, 0x90);
921  // set internal ADC and DAC use the same LRCK clock, ADC LRCK as internal
922  // LRCK
923  res |= i2c_write_reg(ES8388_DACCONTROL21, 0x80);
924  // vroi=0
925  res |= i2c_write_reg(ES8388_DACCONTROL23, 0x00);
926  // no volume
927  setVolume(ES_MODULE_DAC, 0, 0);
928 
929  int lrout = 0;
930  if (AUDIO_HAL_DAC_OUTPUT_LINE2 == dac) {
931  lrout = DAC_OUTPUT_LOUT1 | DAC_OUTPUT_ROUT1;
932  } else if (AUDIO_HAL_DAC_OUTPUT_LINE1 == dac) {
933  lrout = DAC_OUTPUT_LOUT2 | DAC_OUTPUT_ROUT2;
934  } else {
935  lrout = DAC_OUTPUT_LOUT1 | DAC_OUTPUT_LOUT2 | DAC_OUTPUT_ROUT1 |
936  DAC_OUTPUT_ROUT2;
937  }
938  // 0x3c Enable DAC and Enable Lout/Rout/1/2
939  res |= i2c_write_reg(ES8388_DACPOWER, lrout);
940 
941  // ADC
942  res |= i2c_write_reg(ES8388_ADCPOWER, 0xFF);
943  // MIC Left and Right channel PGA gain
944  res |= i2c_write_reg(ES8388_ADCCONTROL1, 0xbb);
945 
946  // 0x00 LINSEL & RINSEL, LIN1/RIN1 as ADC Input; DSSEL,use one DS Reg11;
947  lrout = 0;
948  if (AUDIO_HAL_ADC_INPUT_LINE1 == adc) {
949  lrout = ADC_INPUT_MIC1; //ADC_INPUT_LINPUT1_RINPUT1;
950  } else if (AUDIO_HAL_ADC_INPUT_LINE2 == adc) {
951  lrout = ADC_INPUT_MIC2; //ADC_INPUT_LINPUT2_RINPUT2;
952  } else {
953  lrout = ADC_INPUT_DIFFERENCE;
954  }
955  // / 0x00 LINSEL & RINSEL, LIN1/RIN1 as ADC Input; DSSEL,use one DS Reg11;
956  // DSR, LINPUT1-RINPUT1
957  res |= i2c_write_reg(ES8388_ADCCONTROL2, lrout);
958 
959  res |= i2c_write_reg(ES8388_ADCCONTROL3, 0x02);
960  // Left/Right data, Left/Right justified mode, Bits length, I2S format
961  res |= i2c_write_reg(ES8388_ADCCONTROL4, 0x0d);
962  // ADCFsMode,singel SPEED,RATIO=256
963  res |= i2c_write_reg(ES8388_ADCCONTROL5, 0x02);
964 
965  // ALC for Microphone
966  setVolume(ES_MODULE_ADC, 0, 0); // no volume
967  // Power on ADC, Enable LIN&RIN, Power off MICBIAS, set int1lp to low
968  // power mode
969  res |= i2c_write_reg(ES8388_ADCPOWER, 0x09);
970 
971  /* enable es8388 PA */
972  setPAPower(cfg.is_amplifier_active);
973  LOGI("init,out:%02x, in:%02x", dac, adc);
974  return res == ESP_OK;
975  }

1005  bool start(es_module_t mode) {
1006  LOGD(LOG_METHOD);
1007  esp_err_t res = ESP_OK;
1008  uint8_t prev_data = 0, data = 0;
1009  i2c_read_reg(ES8388_DACCONTROL21, &prev_data);
1010  if (mode == ES_MODULE_LINE) {
1011  // 0x00 audio on LIN1&RIN1, 0x09 LIN2&RIN2 by pass enable
1012  res |= i2c_write_reg(ES8388_DACCONTROL16, 0x09);
1013  // left DAC to left mixer enable and LIN signal to left mixer enable 0db
1014  // : bupass enable
1015  res |= i2c_write_reg(ES8388_DACCONTROL17, 0x50);
1016  // right DAC to right mixer enable and LIN signal
1017  // to right mixer enable 0db : bupass enable
1018  res |= i2c_write_reg(ES8388_DACCONTROL20, 0x50);
1019  // enable adc
1020  res |= i2c_write_reg(ES8388_DACCONTROL21, 0xC0);
1021  } else {
1022  // enable dac
1023  res |= i2c_write_reg(ES8388_DACCONTROL21, 0x80);
1024  }
1025 
1026  i2c_read_reg(ES8388_DACCONTROL21, &data);
1027  if (prev_data != data) {
1028  // start state machine
1029  res |= i2c_write_reg(ES8388_CHIPPOWER, 0xF0);
1030 
1031  // res |= i2c_write_reg(ES8388_CONTROL1, 0x16);
1032  // res |= i2c_write_reg(ES8388_CONTROL2, 0x50);
1033  // start state machine
1034  res |= i2c_write_reg(ES8388_CHIPPOWER, 0x00);
1035  }
1036  if (mode == ES_MODULE_ADC || mode == ES_MODULE_ADC_DAC ||
1037  mode == ES_MODULE_LINE) {
1038  // power up adc and line in
1039  res |= i2c_write_reg(ES8388_ADCPOWER, 0x00);
1040  }
1041  if (mode == ES_MODULE_DAC || mode == ES_MODULE_ADC_DAC ||
1042  mode == ES_MODULE_LINE) {
1043  // power up dac and line out
1044  res |= i2c_write_reg(ES8388_DACPOWER, 0x3c);
1045 
1046  setVoiceMute(false);
1047  LOGD( "start default is mode:%d", mode);
1048  }
1049 
1050  return res == ESP_OK;
