14 // Output I2S data to built-in DAC, no matter the data format is 16bit or 32 bit, the DAC module will only take the 8bits from MSB
15 // so we convet it to a singed 16 bit value
16 static inline uint16_t convert8DAC(int64_t value, int value_bits_per_sample){
17  uint16_t result = value;
18  if (value_bits_per_sample!=16){
19  // convert to 16 bit value
20  result = (value * NumberConverter::maxValue(16) / NumberConverter::maxValue(value_bits_per_sample));
21  }
22  // uint_t positive range
23  result+= 32768;
24  return result;
25 }

372  output_size = size_bytes;
373  for (int j=0;j<size_bytes/2;j++){
374  dst[j] = convert8DAC(data[j], adc_config.bits_per_sample);
375  }
376  } break;
377  case 24: {
378  int24_t *data=(int24_t *)src;
379  output_size = (size_bytes/3) * 2;
