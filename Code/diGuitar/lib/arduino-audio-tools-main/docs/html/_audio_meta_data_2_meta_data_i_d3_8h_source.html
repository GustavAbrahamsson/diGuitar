146 void processTagNotFound(const uint8_t* data, size_t len) {
147 // find tags
148 int pos = findTag("TAG+",(const char*) data, len);
149 if (pos>0){
150 tag_ext = new ID3v1Enhanced();
151 if (tag_ext!=nullptr){
152 if (len-pos>=sizeof(ID3v1Enhanced)){
153 memcpy(tag,data+pos,sizeof(ID3v1Enhanced));
154 processNotify();
155 } else {
156 use_bytes_of_next_write = min(sizeof(ID3v1Enhanced), len-pos);
157 memcpy(tag_ext, data+pos, use_bytes_of_next_write);
158 status = TagFoundPartial;
159 }
160 }
161 } else {
162 pos = findTag("TAG", (const char*) data, len);
163 if (pos>0){
164 tag = new ID3v1();
165 if (tag!=nullptr){
166 if (len-pos>=sizeof(ID3v1)){
167 memcpy(tag,data+pos,sizeof(ID3v1));
168 processNotify();
169 } else {
170 use_bytes_of_next_write = min(sizeof(ID3v1), len-pos);
171 memcpy(tag,data+pos,use_bytes_of_next_write);
172 status = TagFoundPartial;
173
174 }
175 }
176 }
177 }
178
179 // we did not find a full tag we check if the end might start with a tag
180 if (pos==-1){
181 if (data[len-3] == 'T' && data[len-2] == 'A' && data[len-1] == 'G'){
182 strcpy(tag_str,"TAG");
183 status = TagFoundPartial;
184 } else if (data[len-2] == 'T' && data[len-1] == 'A' ){
185 strcpy(tag_str,"TA");
186 status = TagFoundPartial;
187 } else if (data[len-1] == 'T'){
188 strcpy(tag_str,"T");
189 status = TagFoundPartial;
190 }

195 void processPartialTagAtTail(const uint8_t* data, size_t len) {

199 if (strncmp((char*)tag_str, "TAG+", 4)==0){
200 tag_ext = new ID3v1Enhanced();
201 memcpy(tag,tag_str, 4);
202 memcpy(tag,data+len,sizeof(ID3v1Enhanced));
203 processNotify();
204 } else if (strncmp((char*)tag_str,"TAG",3)==0){
205 tag = new ID3v1();
206 memcpy(tag,tag_str, 3);

219 } else if (tag_ext!=nullptr){

228 void processNotify() {
229 if (callback==nullptr) return;
230
231 if (tag_ext!=nullptr){
232 callback(Title, tag_ext->title,strnlen(tag_ext->title,60));
233 callback(Artist, tag_ext->artist,strnlen(tag_ext->artist,60));
234 callback(Album, tag_ext->album,strnlen(tag_ext->album,60));
235 callback(Genre, tag_ext->genre,strnlen(tag_ext->genre,30));
236 delete tag_ext;
237 tag_ext = nullptr;
238 status = TagProcessed;
239 }
240
241 if (tag!=nullptr){
242 callback(Title, tag->title,strnlen(tag->title,30));
243 callback(Artist, tag->artist,strnlen(tag->artist,30));
244 callback(Album, tag->album,strnlen(tag->album,30));
245 uint16_t genre = tag->genre;
246 if (genre < sizeof(genres)){
247 const char* genre_str = genres[genre];
248 callback(Genre, genre_str,strlen(genre_str));
249 }
250 delete tag;
251 tag = nullptr;
252 status = TagProcessed;
253 }

374 // calculate the synch save size
375 uint32_t calcSize(uint8_t chars[4]) {
376 uint32_t byte0 = chars[0];
377 uint32_t byte1 = chars[1];
378 uint32_t byte2 = chars[2];
379 uint32_t byte3 = chars[3];
380 return byte0 << 21 | byte1 << 14 | byte2 << 7 | byte3;
381 }

384 void processTagNotFound(const uint8_t* data, size_t len) {
385
386 // activate tag processing when we have an ID3 Tag
387 if (!tag_active && !tag_processed){
388 int pos = findTag("ID3", (const char*) data, len);
389 if (pos>=0){
390 // fill v2 tag header
391 tag_active = true;
392 // if we have enough data for the header we process it
393 if (len>=pos+sizeof(ID3v2)){
394 memcpy(&tagv2, data+pos, sizeof(ID3v2));
395 end_len = total_len + calcSize(tagv2.size);
396 }
397 }
398 }
399
400 // deactivate tag processing when we are out of range
401 if (end_len>0 && total_len>end_len){
402 tag_active = false;
403 tag_processed = false;
404 }
405
406
407 if (tag_active){
408 // process all tags in current buffer
409 const char* partial_tag = nullptr;
410 for (const char* tag : id3_v2_tags){
411 actual_tag = tag;
412 int tag_pos = findTag(tag, (const char*) data, len);
413
414 if (tag_pos>0 && tag_pos+sizeof(ID3v2Frame)<=len){
415 // setup tag header
416 memmove(&frame_header, data+tag_pos, sizeof(ID3v2FrameString));
417
418 // get tag content
419 if(calcSize(frame_header.size) <= len){
420 int l = min(calcSize(frame_header.size)-1, (uint32_t) 256);
421 memset(result,0,256);
422 strncpy((char*)result, (char*) data+tag_pos+ID3FrameSize, l);
423 if (isAscii(l))
424 processNotify();
425 } else {
426 partial_tag = tag;
427 }
428 }
429 }
430
431 // save partial tag information so that we process the remainder with the next write
432 if (partial_tag!=nullptr){
433 int tag_pos = findTag(partial_tag, (const char*) data, len);
434 memmove(&frame_header, data+tag_pos, sizeof(ID3v2FrameString));
435 int size = min(len - tag_pos, (size_t) calcSize(frame_header.size)-1);
436 strncpy((char*)result, (char*)data+tag_pos+ID3FrameSize, size);
437 use_bytes_of_next_write = size;
438 status = PartialTagAtTail;
439 }
440 }
441
442 total_len += len;
443
444 }

451 if (!isascii(result[j])) return false;
452 }

457 void processPartialTagAtTail(const uint8_t* data, size_t len) {
458 int remainder = calcSize(frame_header.size) - use_bytes_of_next_write;
459 memcpy(result+use_bytes_of_next_write, data, remainder);
460 processNotify();

462 status = TagNotFound;
463 processTagNotFound(data+use_bytes_of_next_write, len-use_bytes_of_next_write);
464 }

478 void processNotify() {
479 if (callback!=nullptr && actual_tag!=nullptr && encodingIsSupported()){
480 if (strncmp(actual_tag,"TALB",4)==0)
481 callback(Title, result,strnlen(result, 256));
482 else if (strncmp(actual_tag,"TOPE",4)==0)
483 callback(Artist, result,strnlen(result, 256));
484 else if (strncmp(actual_tag,"TIT2",4)==0)
485 callback(Album, result,strnlen(result, 256));
486 else if (strncmp(actual_tag,"TCON",4)==0) {
487 if (result[0]=='('){
488 // convert genre id to string
489 int end_pos = strpos((char*)result, ")");
490 if (end_pos>0){
491 // we just use the first entry
492 result[end_pos]=0;
493 size_t idx = atoi(result+1);
494 if (idx>=0 && idx<sizeof(genres)){
495 strncpy((char*)result,genres[idx],256);
496 }
497 }
498 }
499 callback(Genre, result,strnlen(result, 256));
500 }
501 }
