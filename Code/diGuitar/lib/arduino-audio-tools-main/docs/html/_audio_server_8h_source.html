195  void processClient() {
196  //LOGD("processClient");
197  if (client_obj) { // if you get a client,
198  LOGI("New Client."); // print a message out the serial port
199  String currentLine = ""; // make a String to hold incoming data from the client
200  while (client_obj.connected()) { // loop while the client's connected
201  if (client_obj.available()) { // if there's bytes to read from the client,
202  char c = client_obj.read(); // read a byte, then
203  if (c == '\n') { // if the byte is a newline character
204 
205  // if the current line is blank, you got two newline characters in a row.
206  // that's the end of the client HTTP request, so send a response:
207  if (currentLine.length() == 0){
208  sendReplyHeader();
209  sendReplyContent();
210  // break out of the while loop:
211  break;
212  } else { // if you got a newline, then clear currentLine:
213  currentLine = "";
214  }
215  }
216  else if (c != '\r') { // if you got anything else but a carriage return character,
217  currentLine += c; // add it to the end of the currentLine
218  }
219  }
220  }
